Bitcoin Live Price Monitor & Telegram Alert Bot

An automated Python script running 24/7 that fetches real-time crypto prices via the CoinGecko API and pushes instant data alerts to a Telegram Channel when volatility thresholds are met.

Role Automation Engineer
Core Tech Stack Python / Telebot / APIs
Interval Check 300 Seconds (5 Mins)

>_ Production Python Source Code

bitcoin_monitor.py
import requests                               
import telebot                                
import time                                   
                                              
TOKEN = '8801809751:AAHDIDbYZA_lckzUV-fhD_IaMm...' 
CHAT_ID = '8594831118'                        
                                              
bot = telebot.TeleBot(TOKEN)                  
                                              
def get_crypto_price(coin_id):                
    url = f"https://api.coingecko.com/api/v3/simple/price?ids={coin_id}&vs_currencies=usd"
    response = requests.get(url).json()       
    return response[coin_id]['usd']           
                                              
coin = 'bitcoin'                              
last_price = None                             
                                              
print(f"Monitoring {coin}...")                
                                              
while True:                                   
    try:                                      
        current_price = get_crypto_price(coin)
        if last_price is not None:            
            change = ((current_price - last_price) / last_price) * 100
            if abs(change) >= 1.0:            
                msg = f"⚠️ VOLATILITY ALERT: {coin.upper()} has changed by {change:.2s}%! Current Price: ${current_price}"
                bot.send_message(CHAT_ID, msg)
                                              
        last_price = current_price            
        print(f"Current {coin} price: ${current_price}")
    except Exception as e:                    
        print(f"Error: {e}")                  
                                              
    time.sleep(300)
